Querying for Payloads

Query a peer

GET /out/payloads
GET /out/payloads/[originator_uuid]/[app_uid]

Headers

Accept: application/json
Accept-Charset: utf-8
Content-Type: application/json;charset=utf-8

Access Control

Send secret property with request if peer requires

Sending a Request

Net::HTTP.start(uri.host, uri.port) do |http|

  request = Net::HTTP::Get.new uri

  request.add_field('Accept', 'application/json')
  request.add_field('Accept-Charset', 'utf-8')
  request.add_field('Content-Type', 'application/json')

  if self.secret
    request.body = {'secret'=>self.secret}.to_json
  end

  response = http.request(request)
  payloads = JSON.parse response.body

end

Response Codes

200  will include payload or payloads array

400  if client sent invalid JSON in request body

403  if invalid or not provided secret when provided

406  if Accept or Accept-Charset are invalid

415  if Content-Type is invalid

503  if host is overcapacity

Processing Received Payloads

payloads.each do |payload|

  payload.delete 'attributes' if payload.include? 'attributes'

  Casa::Payload.translate_in payload

  Casa::Payload.squash_in payload

  next unless Casa::Payload.filter_in payload

  Casa::Payload.transform_in payload

end

Translation

def translate_in p
  p['original'] = t_in_attr_sec p['original']
  p['attributes'] = t_in_attr_sec p['attributes'] if p.include? 'attributes'
  p['journal'].each_index do |idx|
    p['journal'][idx] = t_in_attr_sec p['attributes']
  end if p.include? 'journal'
end
def t_in_attr_sec section
  ['use', 'require'].each do |type|
    if section.include? type
      new_section_type = {}
      section[type].each do |key, value|
        attr = attribute_for_uuid type, key
        new_section_type[attr ? attr.name : key] = value
      end
      section[type] = new_section_type
    end
  end
  section
end

Squashing

payload['attributes'] = payload['original']

if payload.include? 'journal'
  ['use','require'].each do |type|
    payload['journal'].each do |entry|
      entry[type].each do |key, value|

        if custom_attribute_squash? key
          attribute_squash(key, type, payload, value) 
        else
          payload['attributes'][type][key] = value
        end

      end
    end
  end
end
